// Requires the Servo library
#include <Servo.h>

Servo myServo;

void setup() {
  myServo.attach(9);
}

void loop() {
  for (int pos = 0; pos <= 180; pos += 1) {
    myServo.write(pos);
    delay(15);
  }
  for (int pos = 180; pos >= 0; pos -= 1) {
    myServo.write(pos);
    delay(15);
  }
}

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

  lcd.begin(16, 2);
  lcd.print("S1");
}

void loop() {
  lcd.setCursor(0, 1);
  lcd.print(millis() / 1000);
  delay(500);
}

  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);
}

void loop() {
  setColor(255, 0, 0); // Red
  delay(1000);
  setColor(0, 255, 0); // Green
  delay(1000);
  setColor(0, 0, 255); // Blue
  delay(1000);
}

void setColor(int red, int green, int blue) {
  analogWrite(redPin, red);
  analogWrite(greenPin, green);
  analogWrite(bluePin, blue);
}

const int pirPin = 2; // Connect PIR OUT to Digital Pin 2

void setup() {
  Serial.begin(9600);
  pinMode(pirPin, INPUT); // Set the pin as an input
  Serial.println("S1");
}

void loop() {
  // Read the state of the pin
  int motionState = digitalRead(pirPin); 

  if (motionState == HIGH) {
    // Pin is HIGH when motion is detected
    Serial.println("S1");
  } else {
    // Pin is LOW when no motion
    Serial.println("S1");
  }
  
  // A small delay to keep the serial monitor readable
  delay(500); 
}
